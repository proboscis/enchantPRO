 56     enchant.pro.Sensor = enchant.Class.create(enchant.EventTarget, {
 57     	/**
 58     	 * センサークラス
 59     	 * @constructs
 60     	 */
 61     	initialize: function(className){
 62     		if (enchant.pro[className].instance) {
 63     			return enchant.pro[className].instance;
 64     		}
 65     		if(this instanceof enchant.pro[className]) {
 66     			enchant.pro[className].instance = this;
 67     		} else {
 68     			return new enchant.pro[className];
 69     		}
 70     		enchant.EventTarget.call(this);
 71 	    	var sensorList = EP_SENSOR.getSensor(className).getSensorsList();
 72 	    	if(sensorList.size() > 0){
 73 	    		this._sensor = sensorList.get(0);
 74 	    	}
 75 	    	if(this._sensor){
 76 	    		this._sensor.setCallback("enchant.pro." + className + ".onUpdate");
 77 	    	}
 78 	    	
 79 	    	enchant.pro[className].onUpdate = function(data){
 80 		        if (enchant.pro[className].instance) {
 81 		            var e = new Event('update');
 82 		            e.data = data;
 83 		            enchant.pro[className].instance.dispatchEvent(e);
 84 		        }
 85 	    	};
 86     	},
 87     	/**
 88    		 * センサーが利用可能かを返す
 89    		 * @return {Boolean} 
 90    		 */
 91    		isAvailable: function(){
 92    			return this._sensor;
 93    		},
 94    		/**
 95    		 * センサーからの読み取りを開始
 96    		 */
 97    		start: function(){
 98    			if(this._sensor){
 99     			this._sensor.enable();
100    			}
101    		},
102    		/**
103    		 * センサーからの読み取りを終了
104    		 */
105    		stop: function(){
106    			if(this._sensor){
107    				this._sensor.disable();
108    			}
109    		}
110    	});

375     enchant.Sound.load = function(src, mimetype) {
376         var sound = Object.create(enchant.Sound.prototype);
377         enchant.EventTarget.call(sound);
378         var id = 'enchant-audio' + enchant.Game.instance._soundID++;
379         var player = EP_SOUND.createSoundPlayer(src, id);
380         sound.addEventListener('load', function() {
381             Object.defineProperties(sound, {
382                 currentTime: {
383                     get: function() { return player.getCurrentTime() },
384                     set: function(time) { player.setCurrentTime(time) }
385                 },
386                 volume: {
387                     get: function() { return player.getVolume() },
388                     set: function(volume) { player.setVolume(volume) }
389                 }
390             });
391             sound._element = player;
392             sound.duration = player.getDuration();
393         });
394         return sound;
395     };

399         	_element: { value: this._element.clone() },
400         	duration: { value: this.duration }
401         });
402         enchant.EventTarget.call(clone);
403         return clone;

407 	    var touches = [null, null, null, null, null];
408 	    ['start', 'move', 'end'].forEach(function(type) {
409 	        window['androidTouch' + capitalize(type)] = function(touch) {
410 	        	if (touches[touch.identifier]) {
411 	                touch.target = touches[touch.identifier].target;    
412 	            } else {
413 	                touch.target = document.elementFromPoint(touch.pageX, touch.pageY);
414 	            }
415 	
416 	            if (type == 'end') {
417 	                touches[touch.identifier] = null;
418 	            } else {
419 	                touches[touch.identifier] = touch;
420 	            }
421 	
422 	            var e = document.createEvent('MouseEvent');
423 	            e.emulated = true;
424 	            
425 	            e.initMouseEvent('touch' + type, true, false, window, 0,
426 	                touch.pageX, touch.pageY, touch.pageX, touch.pageY, false, false, true, false, 0, null);
427 	            e.touches=touches.filter(function(i) { return i && i.target == this; }, touch.target);
428 	            e.changedTouches = [touch];
429 	            touch.target.dispatchEvent(e);
430 	        };
431 	    });
