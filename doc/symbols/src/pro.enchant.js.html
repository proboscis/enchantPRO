 51 	enchant.pro.Sprite3DTable = [];
 52     /**
 53      * @scope enchant.pro.Sprite3D.prototype
 54      */
 55     enchant.pro.Sprite3D = enchant.Class.create(enchant.EventTarget, {
 56 	    /**
 57 	     * 3Dポリゴン表示機能を持ったクラス.
 58 	     * <p>デフォルトではenchantPROのロゴのテクスチャが貼られた立方体がロードされる.</p>
 59 	     * <p>{@link enchant.pro.Scene3D}のインスタンスに追加することで, 画面上に表示することができる.
 60 	     * {@link enchant.pro.Sprite3D#vertices}, {@link enchant.pro.Sprite3D#indices},
 61 	     * {@link enchant.pro.Sprite3D#normals}などを変更することで、任意のポリゴンを描画することもでき,
 62 	     * テクスチャなども貼付けることができる.</p>
 63 	     * <p>また、ポリゴンを子として追加することも可能で、子は全て親を基準とした座標系で描画される.</p>
 64 	     * @example
 65 	     *   //シーンの初期化
 66 	     *   var scene = new Scene3D();
 67 	     *   //ポリゴンの初期化
 68 	     *   var sprite = new Sprite3D();
 69 	     *   //ポリゴンをシーンに追加
 70 	     *   scene.addChild(sprite);
 71 	     *
 72 	     * @constructs
 73 	     * @extends enchant.EventTarget
 74 	     */
 75         initialize: function() {
 76             enchant.EventTarget.call(this);
 77 
 78             this._mesh = EP_GL.getFactory().createMesh();
 79             this._meshId = this._mesh.getMeshId();
 80             enchant.pro.Sprite3DTable[this._meshId] = this;
 81             this._x = 0;
 82             this._y = 0;
 83             this._z = 0;
 84             /**
 85              * 子ポリゴン要素の配列.
 86              * この要素に子として追加されているポリゴンの一覧を取得できる.
 87              * 子を追加したり削除したりする場合には、この配列を直接操作せずに,
 88              * {@link enchant.pro.Sprite3D#addChild}や{@link enchant.pro.Sprite3D#removeChild}を利用する.
 89              * @type enchant.pro.Sprite3D[]
 90              * @see enchant.pro.Sprite3D#addChild
 91              * @see enchant.pro.Sprite3D#removeChild
 92              */
 93             this.childNodes = [];
 94             
 95             /**
 96              * このポリゴンが現在追加されているシーンオブジェクト.
 97              * どのシーンにも追加されていないときにはnull.
 98              * @type enchant.pro.Scene3D
 99              * @see enchant.pro.Scene3D#addChild
100              */
101             this.scene = null;
102             
103             /**
104              * ポリゴンの親要素.
105              * 親が存在しない場合にはnull.
106              * @type enchant.pro.Sprite3D|enchant.pro.Scene3D
107              */
108             this.parentNode = null;
109 
110             var listener = function(e) {
111                 for (var i = 0, len = this.childNodes.length; i < len; i++) {
112                     var sprite = this.childNodes[i];
113                     sprite.dispatchEvent(e);
114                 }
115             };
116             this.addEventListener('added', listener);
117             this.addEventListener('removed', listener);
118             this.addEventListener('addedtoscene', listener);
119             this.addEventListener('removedfromscene', listener);
120 
121 			var that = this;
122             var render = function() {
123                 that.dispatchEvent(new enchant.Event('render'));
124             };
125             this.addEventListener('addedtoscene', function() {
126                 render();
127                 enchant.Game.instance.addEventListener('exitframe', render);
128             });
129             this.addEventListener('removedfromscene', function() {
130                 enchant.Game.instance.removeEventListener('exitframe', render);
131             });
132 
133             this.addEventListener('render', function() {
134                 if (this._changedTranslation) {
135                     this._mesh.setTranslationJSON(JSON.stringify([this._x, this._y, this._z]));
136                     this._changedTranslation = false;
137                 }
138                 if (this._changedScale) {
139                     this._mesh.setScaleJSON(JSON.stringify([this._scaleX, this._scaleY, this._scaleZ]));
140                     this._changedScale = false;
141                 }
142             });
143         },
144 	    /**
145 	     * 子ポリゴンを追加する.
146 	     * 追加が完了すると、子ポリゴンに対してaddedイベントが発生する.
147 	     * 親が既にシーンに追加されていた場合には、そのシーンに追加され,
148 	     * addedtosceneイベントが発生する.
149 	     * @param {enchant.pro.Sprite3D} sprite 追加する子ポリゴン.
150 	     * @example
151 	     *   var parent = new Sprite3D();
152 	     *   var child = new Sprite3D();
153 	     *   //ポリゴンを別のポリゴンに子として追加
154 	     *   parent.addChild(child);
155 	     * @see enchant.pro.Sprite3D#removeChild
156 	     * @see enchant.pro.Sprite3D#childNodes
157 	     * @see enchant.pro.Sprite3D#parentNode
158 	     */
159         addChild: function(sprite) {
160             this.childNodes.push(sprite);
161             this._mesh.addChild(sprite._mesh);
162             sprite.parentNode = this;
163             sprite.dispatchEvent(new enchant.Event('added'));
164             if (this.scene) {
165                 sprite.scene = this.scene;
166                 sprite.dispatchEvent(new enchant.Event('addedtoscene'));
167             }
168         },
169 	    /**
170 	     * 指定された子ポリゴンを削除する.
171 	     * 削除が完了すると、子ポリゴンに対してremovedイベントが発生する.
172 	     * シーンに追加されていた場合には, そのシーンからも削除され,
173 	     * removedfromsceneイベントが発生する.
174 	     * @param {enchant.pro.Sprite3D} sprite 削除する子ポリゴン.
175 	     * @example
176 	     *   var scene = new Scene3D();
177 	     *   //sceneの一番目の子を削除
178 	     *   scene.removeChild(scene.childNodes[0]);
179 	     * @see enchant.pro.Sprite3D#addChild
180 	     * @see enchant.pro.Sprite3D#childNodes
181 	     * @see enchant.pro.Sprite3D#parentNode
182 	     */
183         removeChild: function(sprite) {
184             var i;
185             if ((i = this.childNodes.indexOf(sprite)) != -1) {
186                 this.childNodes.splice(i, 1);
187             }
188             this._mesh.removeChild(sprite._mesh);
189             sprite.parentNode = null;
190             sprite.dispatchEvent(new enchant.Event('removed'));
191             if (this.scene) {
192                 sprite.scene = null;
193                 sprite.dispatchEvent(new enchant.Event('removedfromscene'));
194             }
195         },
196         /**
197          * ポリゴンを平行移動する.
198          * 現在表示されている位置から, 各軸に対して指定された分だけ平行移動をする.
199          * @param {Number} x x軸方向の平行移動量
200          * @param {Number} y y軸方向の平行移動量
201          * @param {Number} z z軸方向の平行移動量
202          * @example
203          *   var sprite = new Sprite3D();
204          *   //x軸方向に10, y軸方向に3, z軸方向に-20平行移動
205          *   sprite.translate(10, 3, -20);
206          * @see enchant.pro.Sprite3D#x
207          * @see enchant.pro.Sprite3D#y
208          * @see enchant.pro.Sprite3D#z
209          * @see enchant.pro.Sprite3D#scale
210          */
211         translate: function(x, y, z) {
212             this._x += x;
213             this._y += y;
214             this._z += z;
215             this._changedTranslation = true;
216         },
217         /**
218          * ポリゴンを拡大縮小する.
219          * 現在の拡大率から, 各軸に対して指定された倍率分だけ拡大縮小をする.
220          * @param {Number} x x軸方向の拡大率
221          * @param {Number} y y軸方向の拡大率
222          * @param {Number} z z軸方向の拡大率
223          * @example
224          *   var sprite = new Sprite3D();
225          *   //x軸方向に2.0倍, y軸方向に3.0倍, z軸方向に0.5倍に拡大する
226          *   sprite.scale(2,0, 3.0, 0.5);
227          * @see enchant.pro.Sprite3D#scaleX
228          * @see enchant.pro.Sprite3D#scaleY
229          * @see enchant.pro.Sprite3D#scaleZ
230          * @see enchant.pro.Sprite3D#translate
231          */
232         scale: function(x, y, z) {
233             this._scaleX *= x;
234             this._scaleY *= y;
235             this._scaleZ *= z;
236             this._changedScale = true;
237         },
238         /**
239          * ポリゴンの回転行列.
240          * 配列は長さ16の一次元配列であり, 行優先の4x4行列として解釈される.
241          * @example
242          *   var sprite = new Sprite3D();
243          *   //x軸周りに45度回転
244          *   var rotX = Math.PI() / 4;
245          *   sprite.rotation = [
246 		 *       1, 0, 0, 0,
247 		 *       0, Math.cos(rotX), -Math.sin(rotX), 0,
248 		 *       0, Math.sin(rotX), Math.cos(rotX), 0,
249 		 *       0, 0, 0, 1
250 		 *   ];
251          * @type Number[]
252          */
253         rotation: {
254             get: function() {
255                 return this._rotation;
256             },
257             set: function(rotation) {
258                 this._rotation = rotation;
259                 this._mesh.setRotationJSON(JSON.stringify(rotation));
260             }
261         },
262         /**
263          * ポリゴンに適用する変換行列.
264          * @deprecated
265          * @type Number[]
266          */
267         matrix: {
268             get: function() {
269                 return this._matrix;
270             },
271             set: function(matrix) {
272                 this._matrix = matrix;
273                 this._mesh.setMatrixJSON(JSON.stringify(matrix));
274             }
275         },
276         /**
277          * ポリゴンに貼付けられるテクスチャオブジェクト.
278          * テクスチャのマッピングは{@link enchant.pro.Sprite3D#texCoords}にて設定可能である.
279          * @type enchant.pro.Surface3D
280          * @example
281          *   var sprite = new Sprite3D();
282          *   var surface = new Surface3D();
283          *   surface.url = "texture.png";
284          *   sprite.surface = surface;
285          * @see enchant.pro.Sprite3D#texCoords
286          */
287         surface: {
288             get: function() {
289                 return this._surface;
290             },
291             set: function(surface) {
292                 this._surface = surface;
293                 this._mesh.setMaterial(this._surface._material);
294             }
295         },
296         /**
297          * ポリゴンの名前
298          * @type String
299          */
300         name: {
301             get: function() {
302                 return this._name;
303             },
304             set: function(surface) {
305                 this._mesh.setName(this._name = name);
306             }
307         }
308     });

420      * @type Integer[]

423      *   //頂点配列を代入

428      *       1.0, 1.0, 0.0,  //2番目の頂点(1.0, 1.0, 0.0)
429      *       0.0, 1.0, 0.0   //3番目の頂点(0.0, 1.0, 0.0)

453      * @type Number[]

460      *       1.0, 0.0, 0.0,  //1番目の頂点(1.0, 0.0, 0.0)
461      *       1.0, 1.0, 0.0,  //2番目の頂点(1.0, 1.0, 0.0)

464      *

467      *   sprite.normals = [
468      *       0.0, 0.0, 0.0,  //0番目の頂点の法線ベクトル(0.0, 0.0, 0.0)
469      *       1.0, 0.0, 0.0,  //1番目の頂点の法線ベクトル(1.0, 0.0, 0.0)
470      *       1.0, 1.0, 0.0,  //2番目の頂点の法線ベクトル(1.0, 1.0, 0.0)
471      *       0.0, 1.0, 0.0   //3番目の頂点の法線ベクトル(0.0, 1.0, 0.0)
472      *   ];
473      * @see enchant.pro.Sprite3D#vertices
474      * @see enchant.pro.Sprite3D#indices

479     /**
480      * ポリゴンのテクスチャマッピング配列.
481      * 2つの要素を一組としてuv座標を指定する. 全体の要素数は, 頂点の個数nに対して2nとなる.
482      * 2n, 2n+1番目の要素はそれぞれ, n番目の頂点のテクスチャのu, v座標である.
483      * それぞれの座標のとりうる値は0<=u,v<=1である.
484      * @type Number[]
485      * @example
486      *   var sprite = new Sprite3D();
487      *   var surface = new Surface3D();
488      *   surface.url = "texture.png";
489      *   sprite.surface = surface;
490      *
491      *   //頂点配列を代入
492      *   //データはx, y, z, x, y, z...の順に格納する
493      *   sprite.vertices = [
494      *       0.0, 0.0, 0.0,  //0番目の頂点(0.0, 0.0, 0.0)
495      *       1.0, 0.0, 0.0,  //1番目の頂点(1.0, 0.0, 0.0)
496      *       1.0, 1.0, 0.0,  //2番目の頂点(1.0, 1.0, 0.0)
497      *       0.0, 1.0, 0.0   //3番目の頂点(0.0, 1.0, 0.0)
498      *   ];
499      *
500      *   //uv座標配列を代入
501      *   //データはu, v, u, v...の順に格納する
502      *   sprite.texCoords = [
503      *       0.0, 0.0,  //0番目の頂点のuv座標(0.0, 0.0)
504      *       1.0, 0.0,  //1番目の頂点のuv座標(1.0, 0.0)
505      *       1.0, 1.0,  //2番目の頂点のuv座標(1.0, 1.0)
506      *       0.0, 1.0   //3番目の頂点のuv座標(0.0, 1.0)
